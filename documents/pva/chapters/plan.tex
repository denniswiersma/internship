%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% --- Week counting system --- %
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Create a new counter for the *visual* side of the weeks and set it to 36
\newcounter{myWeekNum}
\setcounter{myWeekNum}{36}

% This command makes sure that the counter does not go past the amount of weeks in a year
\newcommand{\myWeek}{
    \themyWeekNum % Retrieve current value of `myWeekNum`
    \stepcounter{myWeekNum} % Go 1 week further
    \ifnum\themyWeekNum=53 % If we're now at week 53
        \setcounter{myWeekNum}{1} % Forcibly set to week 1
    \fi
}
